# Object inspection

---

## symbol vs enumerable

---

```js
const c = Symbol();
const d = Symbol("d");

const o = { a: 1, [c]: 10 };

// non-enumerable by default
Object.defineProperty(o, "b", { value: 2 });
Object.defineProperty(o, d, { value: 20 });
```

## Access

```js
o.a;
// -> 1
o.b;
// -> 2
o[c];
// -> 10
o[d];
// -> 20
```

---

## hasOwnProperty

```js
o.hasOwnProperty("a");
// -> true
o.hasOwnProperty("b");
// -> true
o.hasOwnProperty(c);
// -> true
o.hasOwnProperty(d);
// -> true
```

---

## keys

```js
Object.keys(o);
// -> ["a"]

Object.entries(o);
// -> [["a", 1]]
```

## getOwn

```js
Object.getOwnPropertyNames(o);
// -> ["a", "b"]
Object.getOwnPropertySymbols(o);
// -> [Symbol(), Symbol(d)]
```

---

## assign

```js
Object.assign({}, o);
// -> {a: 1, Symbol(): 10}

{ ...o };
// -> {a: 1, Symbol(): 10}
```

## JSON

```js
// -> `{"a":1}`
```

# Where to use

## 1. Hidden properties

```js
const ROLE = Symbol();

const createAdmin = name => ({
  name,
  [ROLE]: "admin"
});

const isAdmin = ({ [ROLE]: role }) => role === "admin";
```
